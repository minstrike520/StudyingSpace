 & \left[ (0<| x-c |<\delta)
 \left(
| \sqrt{ x }-\sqrt{ c } | = \frac{| x-c |}{\sqrt{ x }+\sqrt{ c }} < \frac{\delta}{\sqrt{ c }} = \frac{\varepsilon \sqrt{ c }}{\sqrt{ c }} = \varepsilon \right) \right]
\end{aligned}
$$
